User {
  id: string,
  firstName: string,
  lastName: string,
  email: string
}

Team {
  id: string,
  name: string,
  createdByUserId: string,
  memberIds: Array<string>,
  capacity: integer,
  eventId: string
}

UserTeams {
  userId: string,
  teamId: string or null,
  eventId: string
}

Event {
  id: string,
  name: string,
  inviteCode: string,
  maxTeamSize: integer
}
